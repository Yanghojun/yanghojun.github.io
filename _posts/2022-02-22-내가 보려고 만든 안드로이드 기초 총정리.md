- onStart()

  - ActivityлҘј нҸ¬к·ёлқјмҡҙл“ңм—җ ліҙлӮҙ мғҒнҳёмһ‘мҡ©н•  мҲҳ мһҲлҸ„лЎқ мӨҖл№„н•Ё. мҳҲлҘјл“Өм–ҙ мқҙ л©”м„ңл“ңм—җм„ң м•ұмқҙ UIлҘј кҙҖлҰ¬н•ҳлҠ” мҪ”л“ңлҘј мҙҲкё°нҷ”н•Ё
  - Activityк°Җ ліҙм—¬м§Җкё° м§Ғм „м—җ нҳём¶ңлҗЁ

- onPause()

  - мӮ¬мҡ©мһҗк°Җ ActivityлҘј л– лӮҳл©ҙ мқҙ л©”м„ңл“ңлҘј нҳём¶ңн•Ё. Activityк°Җ нҸ¬к·ёлқјмҡҙл“ңм—җ мһҲм§Җ м•ҠкІҢ лҗҗлӢӨлҠ”кұё мқҳлҜён•Ё. (мӮ¬мҡ©мһҗк°Җ л©ҖнӢ°мңҲлҸ„мҡ° лӘЁл“ңл©ҙ м—¬м „нһҲ н‘ңмӢңк°Җ к°ҖлҠҘ)
  - мғҒнғңл°”лҘј лӮҙлҰ¬кұ°лӮҳ кёҙкёүл©”м„ём§Җк°Җ нҢқм—…лҗҳкұ°лӮҳ н•ҳлҠ”л“ұ лӢӨлҘё Activityк°Җ нҷ”л©ҙм—җ лӮҳнғҖлӮҳлҠ” мӢңм җм—җ нҳём¶ңлҗЁ

- onResume()

  - Activityк°Җ нҸ¬к·ёлқјмҡҙл“ңм—җ н‘ңмӢңлҗЁ. м—¬кё°м„ң м–ҙл–Ө мқҙлІӨнҠёк°Җ л°ңмғқн•ҙм„ң нҸ¬м»ӨмҠӨк°Җ ліҖкІҪлҗ  л•Ң к№Ңм§Җ м•ұмқҙ onResume()мғҒнғңм—җ лЁёл¬ҙлҰ„.
  - м „нҷ”к°Җ мҳӨкұ°лӮҳ, Activityк°Җ м „нҷҳлҗҳкұ°лӮҳ н–Ҳмқ„ кІҪмҡ° onResume()мқ„ лІ—м–ҙлӮЁ
  - Activityк°Җ ліҙм—¬м§Җкі  Inputмқ„ л°ӣмқ„ мҲҳ мһҲлҠ” мғҒнғңм—җм„ң нҳём¶ңлҗЁ

### Intent-filter

- кө¬м„ұмҡ”мҶҢлҠ” 3к°Җм§Җ

```kotlin
    private lateinit var myFragment: MyFragment
    private lateinit var mySecondFragment: MySecondFragment

        super.onCreate(savedInstanceState)
        binding = ActivityChangeFragmentBinding.inflate(layoutInflater)

        setContentView(binding.root)

        binding.fragmentChangeBtn1.setOnClickListener {
            myFragment = MyFragment.newInstance("hohohoho", "dfsfs")
            supportFragmentManager.beginTransaction().replace(R.id.activity_change_frame_layout, myFragment).commit()
        }

        binding.fragmentChangeBtn2.setOnClickListener {
            mySecondFragment = MySecondFragment.newInstance("hohohohoasfslkadjf", "Asdfsa")
            supportFragmentManager.beginTransaction().replace(R.id.activity_change_frame_layout, mySecondFragment).commit()
        }
    }
```

```kotlin
class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val et_id = findViewById<EditText>(R.id.et_id)
        val et_pw = findViewById<EditText>(R.id.et_pw)

        // нӮӨліҙл“ңмқҳ м—”н„°к°Җ лҲҢл ёмқ„ л•Ң мқҙлІӨнҠё мІҳлҰ¬н•ҳкё° мң„н•ң мҪ”л“ң

            if(actionId == EditorInfo.IME_ACTION_DONE){// м—”н„°к°Җ лҲҢл ёлӢӨл©ҙ
                Login(v) // м•„лһҳ м •мқҳн•ң н•ЁмҲҳлҘј нҳём¶ңн•ҳлқј
                true     // л°ҳнҷҳ нғҖмһ…мқҖ boolean
            } else {
                false   // л°ҳнҷҳ нғҖмһ…мқҖ boolean
            }
        }
    }

    fun Login(v: View){

        // нӮӨліҙл“ң мҲЁкё°кё°
        var imm = getSystemService(Context.INPUT_METHOD_SERVICE) as InputMethodManager
        imm.hideSoftInputFromWindow(v.windowToken, 0)

        Log.d(TAG, "Login: лЎңк·ёмқё м„ұкіө")
        Toast.makeText(this, "лЎңк·ёмқё м„ұкіө", Toast.LENGTH_LONG).show()
    }
}
```

```kotlin
class MainActivity : AppCompatActivity() {
    lateinit var et_id:EditText
    lateinit var et_pw:EditText

        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        et_id = findViewById<EditText>(R.id.et_id)
        et_pw = findViewById<EditText>(R.id.et_pw)

        // м•„мқҙл””, л№„л°ҖлІҲнҳёк°Җ м ҖмһҘлҗҳм–ҙ мһҲлӢӨл©ҙ л¶Ҳлҹ¬мҳӨкі  к·ёл Үм§Җ м•Ҡмңјл©ҙ л№Ҳм№ёмңјлЎң м„ёнҢ…н•Ё

        et_id.setText(pref.getString("м•„мқҙл””", ""))
        et_pw.setText(pref.getString("л№„л°ҖлІҲнҳё", ""))

        et_pw.setOnEditorActionListener() { v, actionId, event ->
            if(actionId == EditorInfo.IME_ACTION_DONE){
                Login(v)
                true
            } else {
                false
            }
        }
    }

    fun Login(v: View){
        if(et_id.text.toString() == "hojun" && et_pw.text.toString() =="0303"){
            Toast.makeText(this, "лЎңк·ёмқё м„ұкіө", Toast.LENGTH_LONG).show()
            var editor = this.getPreferences(0).edit()
            editor.putString("м•„мқҙл””", "hojun").apply()
            editor.putString("л№„л°ҖлІҲнҳё", "0303").apply()
        }

        else{
            Toast.makeText(this, "лЎңк·ёмқё мӢӨнҢЁ", Toast.LENGTH_LONG).show()
        }

        // нӮӨліҙл“ң мҲЁкё°кё°
        var imm = getSystemService(Context.INPUT_METHOD_SERVICE) as InputMethodManager
        imm.hideSoftInputFromWindow(v.windowToken, 0)
    }


}
```

```kotlin
class MainActivity : AppCompatActivity() {
    lateinit var bt_open_camera:Button
    lateinit var iv_photo:ImageView

    private lateinit var resultLauncher: ActivityResultLauncher<Intent>

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        bt_open_camera = findViewById<Button>(R.id.bt_open_camera)
        iv_photo = findViewById<ImageView>(R.id.iv_photo)

        // Activityм—җм„ң мҳЁ resultлҘј мІҳлҰ¬н•ҳлҠ” Launcher
        // мҪңл°ұмқ„ л“ұлЎқн•ҳлҠ” кІғмһ„
        resultLauncher =
            registerForActivityResult(ActivityResultContracts.StartActivityForResult()) { result ->
                if (result.resultCode == Activity.RESULT_OK) {
                    handleCameraImage(result.data) // Launcherк°Җ м№ҙл©”лқј мҙ¬мҳҒмқ„ м„ұкіөм ҒмңјлЎң л§ҲміӨлӢӨлҠ” resultCodeлҘј л°ӣм•ҳмқ„ кІҪмҡ° мӢӨн–ү
                }
            }
        
        // Launcherм—җ м№ҙл©”лқј мҙ¬мҳҒн•ҳлҠ” Intent ліҙлғ„
        bt_open_camera.setOnClickListener {
            val cameraIntent = Intent(MediaStore.ACTION_IMAGE_CAPTURE)
            resultLauncher.launch(cameraIntent)
        }
    }

    private fun handleCameraImage(intent: Intent?) {
        
        // "data"лқјлҠ” keyлЎң мҳЁ мқҙлҜём§Җ лҚ°мқҙн„°лҘј л°ӣмқҢ
        val bitmap = intent?.extras?.get("data") as Bitmap

        // мқҙлҜём§Җл·°мқҳ мқҙлҜём§Җ ліҖкІҪ
        iv_photo.setImageBitmap(bitmap)
    }

}
```
